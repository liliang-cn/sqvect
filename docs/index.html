Enterprise-Grade Features

Built for production with cutting-edge algorithms and optimization techniques

HNSW Indexing

Hierarchical Navigable Small World graphs for lightning-fast approximate nearest neighbor search with sub-linear complexity.

Advanced Quantization

Product, scalar, and binary quantization for optimal memory usage and blazing-fast searches with configurable precision.

Multi-Index Strategies

Combine HNSW, IVF, and flat indices with intelligent voting, merging, and reranking for optimal performance.

sqvect/example.go 
package main

import (
    "context"
    "github.com/liliang-cn/sqvect/pkg/core"
)

func main() {
    // Initialize SQVect store
    store, err := core.NewSQLiteStore(&core.Config{
        DatabasePath: "vectors.db",
        Dimension:    1536,
        Similarity:   core.CosineSimilarity,
        HNSW: core.HNSWConfig{
            Enabled: true,
            M:       16,
            EfConstruction: 200,
        },
    })
    if err != nil {
        panic(err)
    }
    defer store.Close()

    // Add vectors with metadata
    embedding := &core.Embedding{
        ID:       "doc1",
        Vector:   []float32{0.1, 0.2, 0.3...},
        Content:  "Your document content",
        Metadata: map[string]string{
            "category": "technology",
            "author":   "AI Assistant",
        },
    }
    
    store.Upsert(context.Background(), embedding)
    
    // Search with advanced options
    results, err := store.Search(context.Background(), 
        queryVector, 
        core.SearchOptions{
            TopK: 10,
            Filter: map[string]string{
                "category": "technology",
            },
        })
}
