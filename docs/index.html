sqvect/example.go 
package main

import (
    "context"
    "fmt"

    "github.com/liliang-cn/sqvect/v2/pkg/core"
    "github.com/liliang-cn/sqvect/v2/pkg/sqvect"
)

func main() {
    db, err := sqvect.Open(sqvect.DefaultConfig("rag.db"))
    if err != nil {
        panic(err)
    }
    defer db.Close()

    ctx := context.Background()
    db.Vector().CreateDocument(ctx, &core.Document{ID: "doc1", Title: "Go Guide"})

    _, _ = db.Quick().Add(ctx, []float32{0.1, 0.2, 0.3}, "Go is awesome")

    results, _ := db.Quick().Search(ctx, []float32{0.1, 0.2, 0.25}, 1)
    fmt.Println(results[0].Content)
}
